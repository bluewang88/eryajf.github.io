# Short-Description: Bind-mounts needed running before php-fpm.service

case "$1" in
  start)
        $0 stop 2>/dev/null
        for d in $DIRS; do
                mkdir -p "${CHROOT}${d}"

  stop)
        for d in $DIRS; do

        done

        exit 1
        ;;
